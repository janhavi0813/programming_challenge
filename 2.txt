#include <stdio.h>

int countOccurrences(int arr[], int n, int x)
{
	int res = 0;
	for (int i = 0; i < n; i++)
		if (x == arr[i])
			res++;
	return res;
}

int main()
{
	int arr[] = {0,1,1,0,2,3,4};
	int n = sizeof(arr) / sizeof(arr[0]);
	int x = 1;
	printf("%d", countOccurrences(arr, n, x));
	return 0;
}
